# 改用station_id映射，而不是dest_code

# 1. 建立station_id到entry的映射
entry_by_station = {}
for entry in timetable_entries:
    entry_by_station[entry.station_id] = entry

# 2. 建立dest_code到station_id的映射
destcode_to_station = {}
for station_id, station in rail_info.stationList.items():
    for platform in station.platformList:
        destcode_to_station[platform.dest_code] = station_id

# 3. 遍历path.nodeList，使用path的dest_code
for dest_code in path.nodeList:
    station_id = destcode_to_station[dest_code]
